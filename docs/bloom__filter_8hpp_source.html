586inline BloomFilter::BloomFilter(size_t bytes,
587 unsigned hash_num,
588 std::string hash_fn)
589 : bytes(
590 size_t(std::ceil(double(bytes) / sizeof(uint64_t)) * sizeof(uint64_t)))
591 , array_size(get_bytes() / sizeof(array[0]))
592 , array_bits(array_size * CHAR_BIT)
593 , hash_num(hash_num)
594 , hash_fn(std::move(hash_fn))
595{
596 check_error(bytes == 0, "BloomFilter: memory budget must be >0!");
597 check_error(hash_num == 0, "BloomFilter: number of hash values must be >0!");
598 check_error(hash_num > MAX_HASH_VALUES,
599 "BloomFilter: number of hash values cannot be over 1024!");
600 check_warning(
601 sizeof(uint8_t) != sizeof(std::atomic<uint8_t>),
602 "Atomic primitives take extra memory. BloomFilter will have less than " +
603 std::to_string(bytes) + " for bit array.");
604 array = new std::atomic<uint8_t>[array_size];
605 std::memset((void*)array, 0, array_size * sizeof(array[0]));
606}

608inline void
609BloomFilter::insert(const uint64_t* hashes)
610{
611 for (unsigned i = 0; i < hash_num; ++i) {
612 const auto normalized = hashes[i] % array_bits;
613 array[normalized / CHAR_BIT] |= BIT_MASKS[normalized % CHAR_BIT];
614 }
615}

617inline bool
618BloomFilter::contains(const uint64_t* hashes) const
619{
620 for (unsigned i = 0; i < hash_num; ++i) {
621 const auto normalized = hashes[i] % array_bits;
622 const auto mask = BIT_MASKS[normalized % CHAR_BIT];
623 if (!bool(array[normalized / CHAR_BIT] & mask)) {
624 return false;
625 }
626 }
627 return true;
628}

630inline bool
631BloomFilter::contains_insert(const uint64_t* hashes)
632{
633 uint8_t found = 1;
634 for (unsigned i = 0; i < hash_num; ++i) {
635 const auto normalized = hashes[i] % array_bits;
636 const auto bitpos = normalized % CHAR_BIT;
637 const auto mask = BIT_MASKS[bitpos];
638 found &= ((array[normalized / CHAR_BIT].fetch_or(mask) >> bitpos) & 1);
639 }
640 return bool(found);
641}

643inline uint64_t
644BloomFilter::get_pop_cnt() const
645{
646 uint64_t pop_cnt = 0;
647#pragma omp parallel for default(none) reduction(+ : pop_cnt)
648 for (size_t i = 0; i < array_size; ++i) {
649 pop_cnt += pop_cnt_byte(array[i]);
650 }
651 return pop_cnt;
652}

654inline double
655BloomFilter::get_occupancy() const
656{
657 return double(get_pop_cnt()) / double(array_bits);
658}

660inline double
661BloomFilter::get_fpr() const
662{
663 return std::pow(get_occupancy(), double(hash_num));

666inline std::shared_ptr<cpptoml::table>
667BloomFilter::parse_header(std::ifstream& file, const std::string& magic_string)
668{
669 const std::string magic_with_brackets = std::string("[") + magic_string + "]";
670
671 std::string line;
672 std::getline(file, line);
673 if (line != magic_with_brackets) {
674 log_error(
675 std::string("Magic string does not match (likely version mismatch)\n") +
676 "File magic string:\t" + line + "\n" + "Loader magic string:\t" +
677 magic_with_brackets);
678 std::exit(EXIT_FAILURE); // NOLINT(concurrency-mt-unsafe)
679 }
680
681 /* Read bloom filter line by line until it sees "[HeaderEnd]"
682 which is used to mark the end of the header section and
683 assigns the header to a char array*/
684 std::string toml_buffer(line + '\n');
685 bool header_end_found = false;
686 while (bool(std::getline(file, line))) {
687 toml_buffer.append(line + '\n');
688 if (line == "[HeaderEnd]") {
689 header_end_found = true;
690 break;
691 }
692 }
693 if (!header_end_found) {
694 log_error("Pre-built bloom filter does not have the correct header end.");
695 std::exit(EXIT_FAILURE); // NOLINT(concurrency-mt-unsafe)
696 }
697 for (unsigned i = 0; i < PLACEHOLDER_NEWLINES; i++) {
698 std::getline(file, line);
699 }
700
701 // Send the char array to a stringstream for the cpptoml parser to parse
702 std::istringstream toml_stream(toml_buffer);
703 cpptoml::parser toml_parser(toml_stream);
704 const auto header_config = toml_parser.parse();
705
706 // Obtain header values from toml parser and assign them to class members
707 return header_config->get_table(magic_string);
708}

710inline BloomFilter::BloomFilter(const std::string& path)
711{
712 std::ifstream file(path);
713
714 auto table = parse_header(file, BLOOM_FILTER_MAGIC_HEADER);
715 bytes = *(table->get_as<decltype(bytes)>("bytes"));
716 check_warning(
717 sizeof(uint8_t) != sizeof(std::atomic<uint8_t>),
718 "Atomic primitives take extra memory. BloomFilter will have less than " +
719 std::to_string(bytes) + " for bit array.");
720 array_size = bytes / sizeof(std::atomic<uint8_t>);
721 array_bits = array_size * CHAR_BIT;
722 hash_num = *(table->get_as<decltype(hash_num)>("hash_num"));
723 if (table->contains("hash_fn")) {
724 hash_fn = *(table->get_as<std::string>("hash_fn"));
725 }
726
727 array = new std::atomic<uint8_t>[array_size];
728 file.read((char*)array, std::streamsize(array_size * sizeof(array[0])));
729}

731inline void
732BloomFilter::save(const std::string& path)
733{
734 std::ofstream file(path.c_str(), std::ios::out | std::ios::binary);
735
736 /* Initialize cpptoml root table
737 Note: Tables and fields are unordered
738 Ordering of table is maintained by directing the table
739 to the output stream immediately after completion */
740 auto root = cpptoml::make_table();
741
742 /* Initialize bloom filter section and insert fields
743 and output to ostream */
744 auto header = cpptoml::make_table();
745 header->insert("bytes", get_bytes());
746 header->insert("hash_num", get_hash_num());
747 if (!hash_fn.empty()) {
748 header->insert("hash_fn", get_hash_fn());
749 }
750 root->insert(BLOOM_FILTER_MAGIC_HEADER, header);
751 file << *root << "[HeaderEnd]\n";
752 for (unsigned i = 0; i < PLACEHOLDER_NEWLINES; i++) {
753 if (i == 1) {
754 file << " <binary data>";
755 }
756 file << '\n';
757 }
758
759 file.write((char*)array, std::streamsize(array_size * sizeof(array[0])));
760}

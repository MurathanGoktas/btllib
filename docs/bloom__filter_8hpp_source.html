630 inline bool
631 BloomFilter::contains_insert(const uint64_t* hashes)
632 {
633  uint8_t found = 1;
634  for (unsigned i = 0; i < hash_num; ++i) {
635  const auto normalized = hashes[i] % array_bits;
636  const auto bitpos = normalized % CHAR_BIT;
637  const auto mask = BIT_MASKS[bitpos];
638  found &= ((array[normalized / CHAR_BIT].fetch_or(mask) >> bitpos) & 1);
639  }
640  return bool(found);
641 }

666 inline std::shared_ptr<cpptoml::table>
667 BloomFilter::parse_header(std::ifstream& file, const std::string& magic_string)
668 {
669  const std::string magic_with_brackets = std::string("[") + magic_string + "]";
670 
671  std::string line;
672  std::getline(file, line);
673  if (line != magic_with_brackets) {
674  log_error(
675  std::string("Magic string does not match (likely version mismatch)\n") +
676  "File magic string:\t" + line + "\n" + "Loader magic string:\t" +
677  magic_with_brackets);
678  std::exit(EXIT_FAILURE); // NOLINT(concurrency-mt-unsafe)
679  }
680 
681  /* Read bloom filter line by line until it sees "[HeaderEnd]"
682  which is used to mark the end of the header section and
683  assigns the header to a char array*/
684  std::string toml_buffer(line + '\n');
685  bool header_end_found = false;
686  while (bool(std::getline(file, line))) {
687  toml_buffer.append(line + '\n');
688  if (line == "[HeaderEnd]") {
689  header_end_found = true;
690  break;
691  }
692  }
693  if (!header_end_found) {
694  log_error("Pre-built bloom filter does not have the correct header end.");
695  std::exit(EXIT_FAILURE); // NOLINT(concurrency-mt-unsafe)
696  }
697  for (unsigned i = 0; i < PLACEHOLDER_NEWLINES; i++) {
698  std::getline(file, line);
699  }
700 
701  // Send the char array to a stringstream for the cpptoml parser to parse
702  std::istringstream toml_stream(toml_buffer);
703  cpptoml::parser toml_parser(toml_stream);
704  const auto header_config = toml_parser.parse();
705 
706  // Obtain header values from toml parser and assign them to class members
707  return header_config->get_table(magic_string);
708 }

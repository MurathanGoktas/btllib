655inline BloomFilter::BloomFilter(size_t bytes,
656 unsigned hash_num,
657 std::string hash_fn)
658 : bytes(
659 size_t(std::ceil(double(bytes) / sizeof(uint64_t)) * sizeof(uint64_t)))
660 , array_size(get_bytes() / sizeof(array[0]))
661 , array_bits(array_size * CHAR_BIT)
662 , hash_num(hash_num)
663 , hash_fn(std::move(hash_fn))
664 , array(new std::atomic<uint8_t>[array_size])
665{
666 check_error(bytes == 0, "BloomFilter: memory budget must be >0!");
667 check_error(hash_num == 0, "BloomFilter: number of hash values must be >0!");
668 check_error(hash_num > MAX_HASH_VALUES,
669 "BloomFilter: number of hash values cannot be over 1024!");
670 check_warning(
671 sizeof(uint8_t) != sizeof(std::atomic<uint8_t>),
672 "Atomic primitives take extra memory. BloomFilter will have less than " +
673 std::to_string(bytes) + " for bit array.");
674 std::memset((void*)array.get(), 0, array_size * sizeof(array[0]));
675}

677inline void
678BloomFilter::insert(const uint64_t* hashes)
679{
680 for (unsigned i = 0; i < hash_num; ++i) {
681 const auto normalized = hashes[i] % array_bits;
682 array[normalized / CHAR_BIT] |= BIT_MASKS[normalized % CHAR_BIT];
683 }
684}

686inline bool
687BloomFilter::contains(const uint64_t* hashes) const
688{
689 for (unsigned i = 0; i < hash_num; ++i) {
690 const auto normalized = hashes[i] % array_bits;
691 const auto mask = BIT_MASKS[normalized % CHAR_BIT];
692 if (!bool(array[normalized / CHAR_BIT] & mask)) {
693 return false;
694 }
695 }
696 return true;
697}

699inline bool
700BloomFilter::contains_insert(const uint64_t* hashes)
701{
702 uint8_t found = 1;
703 for (unsigned i = 0; i < hash_num; ++i) {
704 const auto normalized = hashes[i] % array_bits;
705 const auto bitpos = normalized % CHAR_BIT;
706 const auto mask = BIT_MASKS[bitpos];
707 found &= ((array[normalized / CHAR_BIT].fetch_or(mask) >> bitpos) & 1);
708 }
709 return bool(found);
710}

712inline uint64_t
713BloomFilter::get_pop_cnt() const
714{
715 uint64_t pop_cnt = 0;
716#pragma omp parallel for default(none) reduction(+ : pop_cnt)
717 for (size_t i = 0; i < array_size; ++i) {
718 pop_cnt += pop_cnt_byte(array[i]);
719 }
720 return pop_cnt;
721}

723inline double
724BloomFilter::get_occupancy() const
725{
726 return double(get_pop_cnt()) / double(array_bits);
727}

729inline double
730BloomFilter::get_fpr() const
731{
732 return std::pow(get_occupancy(), double(hash_num));

745inline std::shared_ptr<cpptoml::table>
746BloomFilterInitializer::parse_header(std::ifstream& file,
747 const std::string& expected_signature)
748{
749 std::string file_signature;
750 if (!check_file_signature(file, file_signature, expected_signature)) {
751 log_error(std::string(
752 "File signature does not match (possibly version mismatch)\n") +
753 "File signature: \t" + file_signature + "\n" +
754 "Expected signature:\t" + expected_signature);
755 std::exit(EXIT_FAILURE); // NOLINT(concurrency-mt-unsafe)
756 }
757
758 /* Read bloom filter line by line until it sees "[HeaderEnd]"
759 which is used to mark the end of the header section and
760 assigns the header to a char array*/
761 std::string toml_buffer(file_signature + '\n');
762 std::string line;
763 bool header_end_found = false;
764 while (bool(std::getline(file, line))) {
765 toml_buffer.append(line + '\n');
766 if (line == "[HeaderEnd]") {
767 header_end_found = true;
768 break;
769 }
770 }
771 if (!header_end_found) {
772 log_error("Pre-built bloom filter does not have the correct header end.");
773 std::exit(EXIT_FAILURE); // NOLINT(concurrency-mt-unsafe)
774 }
775 for (unsigned i = 0; i < PLACEHOLDER_NEWLINES; i++) {
776 std::getline(file, line);
777 }
778
779 // Send the char array to a stringstream for the cpptoml parser to parse
780 std::istringstream toml_stream(toml_buffer);
781 cpptoml::parser toml_parser(toml_stream);
782 const auto header_config = toml_parser.parse();
783
784 // Obtain header values from toml parser and assign them to class members
785 const auto header_string =
786 file_signature.substr(1, file_signature.size() - 2); // Remove [ ]
787 return header_config->get_table(header_string);
788}

828 }

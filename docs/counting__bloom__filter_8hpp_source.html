379 template<typename T>
380 inline void
381 CountingBloomFilter<T>::insert(const uint64_t* hashes)
382 {
383  // Update flag to track if increment is done on at least one counter
384  bool update_done = false;
385  T new_val;
386  T min_val = contains(hashes);
387  while (!update_done) {
388  // Simple check to deal with overflow
389  new_val = min_val + 1;
390  if (min_val > new_val) {
391  return;
392  }
393  for (size_t i = 0; i < hash_num; ++i) {
394  decltype(min_val) temp_min_val = min_val;
395  if (array[hashes[i] % array_size].compare_exchange_strong(temp_min_val,
396  new_val)) {
397  update_done = true;
398  }
399  }
400  // Recalculate minval because if increment fails, it needs a new minval to
401  // use and if it doesnt hava a new one, the while loop runs forever.
402  if (!update_done) {
403  min_val = contains(hashes);
404  }
405  }
406 }

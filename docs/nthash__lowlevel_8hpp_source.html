34inline uint64_t
35srol(const uint64_t x)
36{
37 uint64_t m = ((x & 0x8000000000000000ULL) >> 30) | // NOLINT
38 ((x & 0x100000000ULL) >> 32); // NOLINT
39 return ((x << 1) & 0xFFFFFFFDFFFFFFFFULL) | m; // NOLINT
40}

51inline uint64_t
52srol(const uint64_t x, const unsigned d)
53{
54 uint64_t v = (x << d) | (x >> (64 - d)); // NOLINT
55 uint64_t y = (v ^ (v >> 33)) & // NOLINT
56 (std::numeric_limits<uint64_t>::max() >> (64 - d)); // NOLINT
57 return v ^ (y | (y << 33)); // NOLINT
58}

68inline uint64_t
69sror(const uint64_t x)
70{
71 uint64_t m = ((x & 0x200000000ULL) << 30) | ((x & 1ULL) << 32); // NOLINT
72 return ((x >> 1) & 0xFFFFFFFEFFFFFFFFULL) | m; // NOLINT
73}

83inline uint64_t
84ntf64(const char* kmer_seq, const unsigned k)
85{
86 uint64_t h_val = 0;
87 for (unsigned i = 0; i < k / 4; i++) {
88 h_val = srol(h_val, 4);
89 uint8_t curr_offset = 4 * i;
90 uint8_t tetramer_loc =
91 64 * CONVERT_TAB[(unsigned char)kmer_seq[curr_offset]] + // NOLINT
92 16 * CONVERT_TAB[(unsigned char)kmer_seq[curr_offset + 1]] + // NOLINT
93 4 * CONVERT_TAB[(unsigned char)kmer_seq[curr_offset + 2]] +
94 CONVERT_TAB[(unsigned char)kmer_seq[curr_offset + 3]];
95 h_val ^= TETRAMER_TAB[tetramer_loc];
96 }
97 unsigned remainder = k % 4;
98 h_val = srol(h_val, remainder);
99 if (remainder == 3) {
100 uint8_t trimer_loc =
101 16 * CONVERT_TAB[(unsigned char)kmer_seq[k - 3]] + // NOLINT
102 4 * CONVERT_TAB[(unsigned char)kmer_seq[k - 2]] +
103 CONVERT_TAB[(unsigned char)kmer_seq[k - 1]];
104 // cppcheck-suppress arrayIndexOutOfBounds
105 h_val ^= TRIMER_TAB[trimer_loc];
106 } else if (remainder == 2) {
107 uint8_t dimer_loc = 4 * CONVERT_TAB[(unsigned char)kmer_seq[k - 2]] +
108 CONVERT_TAB[(unsigned char)kmer_seq[k - 1]];
109 // cppcheck-suppress arrayIndexOutOfBounds
110 h_val ^= DIMER_TAB[dimer_loc];
111 } else if (remainder == 1) {
112 h_val ^= SEED_TAB[(unsigned char)kmer_seq[k - 1]];
113 }
114 return h_val;
115}

126inline uint64_t
127ntr64(const char* kmer_seq, const unsigned k)
128{
129 uint64_t h_val = 0;
130 unsigned remainder = k % 4;
131 if (remainder == 3) {
132 uint8_t trimer_loc =
133 16 * RC_CONVERT_TAB[(unsigned char)kmer_seq[k - 1]] + // NOLINT
134 4 * RC_CONVERT_TAB[(unsigned char)kmer_seq[k - 2]] +
135 RC_CONVERT_TAB[(unsigned char)kmer_seq[k - 3]];
136 // cppcheck-suppress arrayIndexOutOfBounds
137 h_val ^= TRIMER_TAB[trimer_loc];
138 } else if (remainder == 2) {
139 uint8_t dimer_loc = 4 * RC_CONVERT_TAB[(unsigned char)kmer_seq[k - 1]] +
140 RC_CONVERT_TAB[(unsigned char)kmer_seq[k - 2]];
141 // cppcheck-suppress arrayIndexOutOfBounds
142 h_val ^= DIMER_TAB[dimer_loc];
143 } else if (remainder == 1) {
144 h_val ^= SEED_TAB[(unsigned char)kmer_seq[k - 1] & CP_OFF];
145 }
146 for (unsigned i = 0; i < k / 4; i++) {
147 h_val = srol(h_val, 4);
148 uint8_t curr_offset = 4 * (k / 4 - i) - 1;
149 uint8_t tetramer_loc =
150 64 * RC_CONVERT_TAB[(unsigned char)kmer_seq[curr_offset]] + // NOLINT
151 16 * RC_CONVERT_TAB[(unsigned char)kmer_seq[curr_offset - 1]] + // NOLINT
152 4 * RC_CONVERT_TAB[(unsigned char)kmer_seq[curr_offset - 2]] +
153 RC_CONVERT_TAB[(unsigned char)kmer_seq[curr_offset - 3]];
154 h_val ^= TETRAMER_TAB[tetramer_loc];
155 }
156 return h_val;
157}

342inline void
343nte64(const uint64_t bh_val,
344 const unsigned k,
345 const unsigned h,
346 uint64_t* h_val)
347{
348 uint64_t t_val;
349 h_val[0] = bh_val;
350 for (unsigned i = 1; i < h; i++) {
351 t_val = bh_val * (i ^ k * MULTISEED);
352 t_val ^= t_val >> MULTISHIFT;
353 h_val[i] = t_val;
354 }

687inline uint64_t
688mask_hash(uint64_t& fk_val,
689 uint64_t& rk_val,
690 const char* seed_seq,
691 const char* kmer_seq,
692 const unsigned k)
693{
694 uint64_t fs_val = fk_val, rs_val = rk_val;
695 for (unsigned i = 0; i < k; i++) {
696 if (seed_seq[i] != '1') {
697 fs_val ^= MS_TAB((unsigned char)kmer_seq[i], k - 1 - i);
698 rs_val ^= MS_TAB((unsigned char)kmer_seq[i] & CP_OFF, i);
699 }
700 }
701 return CANONICAL(fs_val, rs_val);
702}

718inline void
719sub_hash(uint64_t fh_val,
720 uint64_t rh_val,
721 const char* kmer_seq,
722 const std::vector<unsigned>& positions,
723 const std::vector<unsigned char>& new_bases,
724 const unsigned k,
725 const unsigned m,
726 uint64_t* h_val)
727{
728 uint64_t b_val = 0;
729
730 for (size_t i = 0; i < positions.size(); i++) {
731 const auto pos = positions[i];
732 const auto new_base = new_bases[i];
733
734 fh_val ^= MS_TAB((unsigned char)kmer_seq[pos], k - 1 - pos);
735 fh_val ^= MS_TAB(new_base, k - 1 - pos);
736
737 rh_val ^= MS_TAB((unsigned char)kmer_seq[pos] & CP_OFF, pos);
738 rh_val ^= MS_TAB(new_base & CP_OFF, pos);
739 }
740
741 b_val = CANONICAL(fh_val, rh_val);
742 nte64(b_val, k, m, h_val);
743}

769inline bool
770ntmsm64(const char* kmer_seq,
771 const std::vector<SpacedSeed>& seed_seq,
772 const std::vector<std::vector<unsigned>>& monomers,
773 const unsigned k,
774 const unsigned m,
775 const unsigned m2,
776 uint64_t* fh_nomonos,
777 uint64_t* rh_nomonos,
778 uint64_t* fh_val,
779 uint64_t* rh_val,
780 unsigned& loc_n,
781 uint64_t* h_val)
782{
783 uint64_t fh_seed, rh_seed;
784 unsigned i_base, block_start, block_end;
785 const SpacedSeed* seed = nullptr;
786 for (unsigned i_seed = 0; i_seed < m; i_seed++) {
787 seed = &seed_seq[i_seed];
788 fh_seed = 0;
789 rh_seed = 0;
790 for (int i_block = 0; i_block < (int)seed->size() - 1; i_block += 2) {
791 // cppcheck-suppress arrayIndexOutOfBounds
792 // cppcheck-suppress stlOutOfBounds
793 block_start = seed->at(i_block);
794 // cppcheck-suppress arrayIndexOutOfBounds
795 // cppcheck-suppress stlOutOfBounds
796 block_end = seed->at(i_block + 1);
797 for (unsigned pos = block_start; pos < block_end; pos++) {
798 if (kmer_seq[pos] == SEED_N) {
799 loc_n = pos;
800 return false;
801 }
802 fh_seed ^= MS_TAB((unsigned char)kmer_seq[pos], k - 1 - pos);
803 rh_seed ^= MS_TAB((unsigned char)kmer_seq[pos] & CP_OFF, pos);
804 }
805 }
806 fh_nomonos[i_seed] = fh_seed;
807 rh_nomonos[i_seed] = rh_seed;
808 for (unsigned pos : monomers[i_seed]) {
809 fh_seed ^= MS_TAB((unsigned char)kmer_seq[pos], k - 1 - pos);
810 rh_seed ^= MS_TAB((unsigned char)kmer_seq[pos] & CP_OFF, pos);
811 }
812 fh_val[i_seed] = fh_seed;
813 rh_val[i_seed] = rh_seed;
814 i_base = i_seed * m2;
815 h_val[i_base] = CANONICAL(fh_seed, rh_seed);
816 for (unsigned i_hash = 1; i_hash < m2; i_hash++) {
817 h_val[i_base + i_hash] = h_val[i_base] * (i_hash ^ k * MULTISEED);
818 h_val[i_base + i_hash] ^= h_val[i_base + i_hash] >> MULTISHIFT;
819 }
820 }
821 return true;
822}

824#define NTMSM64(ROL_HANDLING, IN_HANDLING, OUT_HANDLING, ROR_HANDLING) \
825 unsigned char char_out, char_in; \
826 uint64_t fh_seed, rh_seed; \
827 unsigned i_out, i_in, i_base; \
828 const SpacedSeed* seed = nullptr; \
829 for (unsigned i_seed = 0; i_seed < m; i_seed++) { \
830 seed = &seed_seq[i_seed]; \
831 ROL_HANDLING /* NOLINT(bugprone-macro-parentheses) */ \
832 for (int i_block = 0; i_block < (int)seed->size() - 1; i_block += 2) \
833 { \
834 /* cppcheck-suppress arrayIndexOutOfBounds */ \
835 /* cppcheck-suppress stlOutOfBounds */ \
836 IN_HANDLING \
837 /* cppcheck-suppress arrayIndexOutOfBounds */ \
838 /* cppcheck-suppress stlOutOfBounds */ \
839 OUT_HANDLING \
840 fh_seed ^= MS_TAB(char_out, k - i_out); \
841 fh_seed ^= MS_TAB(char_in, k - i_in); \
842 rh_seed ^= MS_TAB(char_out & CP_OFF, i_out); \
843 rh_seed ^= MS_TAB(char_in & CP_OFF, i_in); \
844 } \
845 ROR_HANDLING /* NOLINT(bugprone-macro-parentheses) */ \
846 fh_nomonos[i_seed] = fh_seed; \
847 rh_nomonos[i_seed] = rh_seed; \
848 for (unsigned pos : monomers[i_seed]) { \
849 fh_seed ^= MS_TAB((unsigned char)kmer_seq[pos + 1], k - 1 - pos); \
850 rh_seed ^= MS_TAB((unsigned char)kmer_seq[pos + 1] & CP_OFF, pos); \
851 } \
852 fh_val[i_seed] = fh_seed; \
853 rh_val[i_seed] = rh_seed; \
854 i_base = i_seed * m2; \
855 h_val[i_base] = CANONICAL(fh_seed, rh_seed); \
856 for (unsigned i_hash = 1; i_hash < m2; i_hash++) { \
857 h_val[i_base + i_hash] = h_val[i_base] * (i_hash ^ k * MULTISEED); \
858 h_val[i_base + i_hash] ^= h_val[i_base + i_hash] >> MULTISHIFT; \
859 } \
860 }

967inline void
968ntmsm64(const char* kmer_seq,
969 const char in,
970 const std::vector<SpacedSeed>& seed_seq,
971 const std::vector<std::vector<unsigned>>& monomers,
972 const unsigned k,
973 const unsigned m,
974 const unsigned m2,
975 uint64_t* fh_nomonos,
976 uint64_t* rh_nomonos,
977 uint64_t* fh_val,
978 uint64_t* rh_val,
979 uint64_t* h_val)
980{
981 NTMSM64(
982 fh_seed = srol(fh_nomonos[i_seed]); rh_seed = rh_nomonos[i_seed];
983 , i_in = seed->at(i_block + 1);
984 if (i_in > k - 1) { char_in = in; } else {
985 char_in = (unsigned char)kmer_seq[i_in];
986 },
